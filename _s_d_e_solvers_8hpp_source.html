273 EulerMaruyama::stepForward(vectorField *field, vectorFieldStochastic *stocField,
274  gsl_vector *currentState)
275 {
276  gsl_vector_view tmp = gsl_matrix_row(work, 0);
277  gsl_vector_view tmp1 = gsl_matrix_row(work, 1);
278 
279  // Evalueate fields
280  field->evalField(currentState, &tmp.vector);
281  stocField->evalField(currentState, &tmp1.vector);
282 
283  // Add drift
284  gsl_vector_scale(&tmp.vector, dt);
285  gsl_vector_add(currentState, &tmp.vector);
286 
287  // Add diffusion
288  gsl_vector_scale(&tmp1.vector, sqrt(dt));
289  gsl_vector_add(currentState, &tmp1.vector);
290 
291  return;
