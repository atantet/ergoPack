894  gsl_vector *sum;

896  double eps;
897 
900  rowOut = gsl_vector_uint_calloc(M->size1);
901  for (n = 0; n < M->size1; n++)
902  {
903  if (gsl_vector_get(rowCut, n) < tol)
904  {
905  gsl_vector_uint_set(rowOut, n, 1);
906  gsl_vector_set(rowCut, n, 0.);
907  }
908  }
909 
910  colOut = gsl_vector_uint_calloc(M->size2);
911  for (n = 0; n < M->size2; n++)
912  {
913  if (gsl_vector_get(colCut, n) < tol)
914  {
915  gsl_vector_uint_set(colOut, n, 1);
916  gsl_vector_set(colCut, n, 0.);
917  }
918  }
919 
920  if (GSL_SPMATRIX_ISTRIPLET(M))
921  {
922  GSL_ERROR("first index out of range", GSL_EINVAL);
923  }
924  else if (GSL_SPMATRIX_ISCCS(M))
925  {
926  for (outerIdx = 0; outerIdx < M->size2; outerIdx++)
927  {
928  isColOut = gsl_vector_uint_get(colOut, outerIdx);
929  for (p = M->p[outerIdx]; p < M->p[outerIdx + 1]; p++)
930  {
931  isRowOut = gsl_vector_uint_get(rowOut, M->i[p]);
932  // Remove elements of states to be removed
933  if (isRowOut || isColOut)
934  M->data[p] = 0.;
935  }
936  }
937  }
938  else if (GSL_SPMATRIX_ISCRS(M))
939  {
940  for (outerIdx = 0; outerIdx < M->size1; outerIdx++)
941  {
942  isRowOut = gsl_vector_uint_get(rowOut, outerIdx);
943  for (p = M->p[outerIdx]; p < M->p[outerIdx + 1]; p++)
944  {
945  isColOut = gsl_vector_uint_get(colOut, M->i[p]);
946  // Remove elements of states to be removed
947  if (isRowOut || isColOut)
948  M->data[p] = 0.;
949  }
950  }
951  }
952 
954  switch (norm)
955  {
956  case 0:
957  totalSum = gsl_spmatrix_get_sum(M);
958  gsl_spmatrix_scale(M, 1. / totalSum);
959  break;
960  case 1:
961  eps = std::numeric_limits<double>::epsilon() * M->size2 * 100;
962  sum = gsl_vector_alloc(M->size1);
963  gsl_spmatrix_get_rowsum(sum, M);
964  gsl_spmatrix_div_rows(M, sum, eps);
965  gsl_vector_free(sum);
966  break;
967  case 2:
968  eps = std::numeric_limits<double>::epsilon() * M->size1 * 100;
969  sum = gsl_vector_alloc(M->size1);
970  gsl_spmatrix_get_colsum(sum, M);
971  gsl_spmatrix_div_cols(M, sum, eps);
972  gsl_vector_free(sum);
973  break;
974  default:
975  break;
976  }
977  gsl_vector_normalize(rowCut);
978  gsl_vector_normalize(colCut);
979 
981  gsl_vector_uint_free(rowOut);
982  gsl_vector_uint_free(colOut);
