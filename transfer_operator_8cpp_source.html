555 {

557  size_t nTraj;
558  size_t NFilled;
559  size_t box0, boxf;
560 
561  if (gridMem)
562  {
563  nTraj = gridMem->size1;
564 
565  // Initialized all boxes as empty (as marked by N)
566  gsl_vector_uint_set_all(mask, N);
567 
568  // Loop over the states to check which box is filled
569  for (size_t traj = 0; traj < nTraj; traj++)
570  {
572  box0 = gsl_matrix_uint_get(gridMem, traj, 0);
573  boxf = gsl_matrix_uint_get(gridMem, traj, 1);
574 
575  // Flag initial and final boxes only if both are found
576  // (otherwise the transition will not be count by getTransitionCountTriplet)n
577  if ((box0 < N) && (boxf < N))
578  {
579  gsl_vector_uint_set(mask, box0, 1);
580  gsl_vector_uint_set(mask, boxf, 1);
581  }
582  }
583 
584  // Loop over the boxes to finish the mask
585  NFilled = 0;
586  for (size_t box = 0; box < N; box++)
587  {
588  if (gsl_vector_uint_get(mask, box) == 1)
589  {
590  gsl_vector_uint_set(mask, box, NFilled);
591  NFilled++;
592  }
593  }
594  }
595  else
596  {
597  for (size_t box = 0; box < N; box++)
598  gsl_vector_uint_set(mask, box, box);
599  NFilled = N;
600  }

621 filterStochasticMatrix(gsl_spmatrix *M,
622  gsl_vector *rowCut, gsl_vector *colCut,
623  double tol, int norm)
624 {
625  size_t outerIdx, p, n;
626  size_t isRowOut, isColOut;
627  double totalSum;
628  gsl_vector *sum;
629  gsl_vector_uint *rowOut, *colOut;
630  double eps;
631 
634  rowOut = gsl_vector_uint_calloc(M->size1);
635  for (n = 0; n < M->size1; n++)
636  {
637  if (gsl_vector_get(rowCut, n) < tol)
638  {
639  gsl_vector_uint_set(rowOut, n, 1);
640  gsl_vector_set(rowCut, n, 0.);
641  }
642  }
643 
644  colOut = gsl_vector_uint_calloc(M->size2);
645  for (n = 0; n < M->size2; n++)
646  {
647  if (gsl_vector_get(colCut, n) < tol)
648  {
649  gsl_vector_uint_set(colOut, n, 1);
650  gsl_vector_set(colCut, n, 0.);
651  }
652  }
653 
654  if (GSL_SPMATRIX_ISTRIPLET(M))
655  {
656  GSL_ERROR("first index out of range", GSL_EINVAL);
657  }
658  else if (GSL_SPMATRIX_ISCCS(M))
659  {
660  for (outerIdx = 0; outerIdx < M->size2; outerIdx++)
661  {
662  isColOut = gsl_vector_uint_get(colOut, outerIdx);
663  for (p = M->p[outerIdx]; p < M->p[outerIdx + 1]; p++)
664  {
665  isRowOut = gsl_vector_uint_get(rowOut, M->i[p]);
666  // Remove elements of states to be removed
667  if (isRowOut || isColOut)
668  M->data[p] = 0.;
669  }
670  }
671  }
672  else if (GSL_SPMATRIX_ISCRS(M))
673  {
674  for (outerIdx = 0; outerIdx < M->size1; outerIdx++)
675  {
676  isRowOut = gsl_vector_uint_get(rowOut, outerIdx);
677  for (p = M->p[outerIdx]; p < M->p[outerIdx + 1]; p++)
678  {
679  isColOut = gsl_vector_uint_get(colOut, M->i[p]);
680  // Remove elements of states to be removed
681  if (isRowOut || isColOut)
682  M->data[p] = 0.;
683  }
684  }
685  }
686 
688  switch (norm)
689  {
690  case 0:
691  totalSum = gsl_spmatrix_get_sum(M);
692  gsl_spmatrix_scale(M, 1. / totalSum);
693  break;
694  case 1:
695  eps = std::numeric_limits<double>::epsilon() * M->size2 * 100;
696  sum = gsl_vector_alloc(M->size1);
697  gsl_spmatrix_get_rowsum(sum, M);
698  gsl_spmatrix_div_rows(M, sum, eps);
699  gsl_vector_free(sum);
700  break;
701  case 2:
702  eps = std::numeric_limits<double>::epsilon() * M->size1 * 100;
703  sum = gsl_vector_alloc(M->size1);
704  gsl_spmatrix_get_colsum(sum, M);
705  gsl_spmatrix_div_cols(M, sum, eps);
706  gsl_vector_free(sum);
707  break;
708  default:
709  break;
710  }
711  gsl_vector_normalize(rowCut);
712  gsl_vector_normalize(colCut);
713 
715  gsl_vector_uint_free(rowOut);
716  gsl_vector_uint_free(colOut);
717 
718  return 0;
719 }
