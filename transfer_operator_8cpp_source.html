765 size_t
766 getMask(const gsl_matrix_uint *gridMem, gsl_vector_uint *mask)
767 {
768  const size_t N = mask->size;
769  const size_t nTraj = gridMem->size1;
770  size_t NFilled;
771  size_t box0, boxf;
772 
773  // Initialized all boxes as empty (as marked by N)
774  gsl_vector_uint_set_all(mask, N);
775 
776  // Loop over the states to check which box is filled
777  for (size_t traj = 0; traj < nTraj; traj++)
778  {
780  box0 = gsl_matrix_uint_get(gridMem, traj, 0);
781  boxf = gsl_matrix_uint_get(gridMem, traj, 1);
782 
783  // Flag initial and final boxes only if both are found
784  // (otherwise the transition will not be count by getTransitionCountTriplet)n
785  if ((box0 < N) && (boxf < N))
786  {
787  gsl_vector_uint_set(mask, box0, 1);
788  gsl_vector_uint_set(mask, boxf, 1);
789  }
790  }
791 
792  // Loop over the boxes to finish the mask
793  NFilled = 0;
794  for (size_t box = 0; box < N; box++)
795  {
796  if (gsl_vector_uint_get(mask, box) == 1)
797  {
798  gsl_vector_uint_set(mask, box, NFilled);
799  NFilled++;
800  }
801  }
802 
803  return NFilled;
804 }

893  gsl_spmatrix_scale(M, 1. / totalSum);
894  break;
