131 void
132 RegularGrid::getRegularGrid(const gsl_vector *gridLimitsLow,
133  const gsl_vector *gridLimitsUp)
134 {
135  double delta;
136 
137  // Build uniform grid bounds
138  for (size_t d = 0; d < dim; d++)
139  {
140  // Get spatial step
141  delta = (gsl_vector_get(gridLimitsUp, d) - gsl_vector_get(gridLimitsLow, d))
142  / gsl_vector_uint_get(nx, d);
143 
144  // Set grid bounds
145  gsl_vector_set((*bounds)[d], 0, gsl_vector_get(gridLimitsLow, d));
146  for (size_t i = 1; i < gsl_vector_uint_get(nx, d) + 1; i++)
147  {
148  gsl_vector_set((*bounds)[d], i,
149  gsl_vector_get((*bounds)[d], i-1) + delta);
150  }
151  }
152

165 RegularGrid::getAdaptedRegularGrid(const gsl_vector *nSTDLow,
166  const gsl_vector *nSTDUp,
167  const gsl_matrix *states)
168 {
169  gsl_vector *gridLimitsLow, *gridLimitsUp, *statesMean, *statesSTD;
170 
171  gridLimitsLow = gsl_vector_alloc(dim);
172  gridLimitsUp = gsl_vector_alloc(dim);
173 
174  // If then number of STD is given, get mean and std of time series
175  // Otherwise, get minimum and maximum
176  if (nSTDLow && nSTDUp)
177  {
178  // Find mean and std
179  statesMean = gsl_vector_alloc(states->size2);
180  statesSTD = gsl_vector_alloc(states->size2);
181  gsl_matrix_get_mean(statesMean, states, 0);
182  gsl_matrix_get_std(statesSTD, states, 0);
183 
184  // Define limits as multiples of std around the mean
185  for (size_t d = 0; d < dim; d++)
186  {
187  gsl_vector_set(gridLimitsLow, d, gsl_vector_get(statesMean, 0)
188  - gsl_vector_get(nSTDLow, d)
189  * gsl_vector_get(statesSTD, d));
190  gsl_vector_set(gridLimitsUp, d, gsl_vector_get(statesMean, d)
191  + gsl_vector_get(nSTDUp, d)
192  * gsl_vector_get(statesSTD, d));
193  }
194 
195  // Free
196  gsl_vector_free(statesMean);
197  gsl_vector_free(statesSTD);
198  }
199  else
200  {
201  // Get min and max of time series
202  gsl_matrix_get_min(gridLimitsLow, states, 0);
203  gsl_matrix_get_max(gridLimitsUp, states, 0);
204  // The upper bound uses a strict inequality,
205  // make sure the maximum is included
206  gsl_vector_add_constant(gridLimitsUp, 1.e-12);
207  }
208 
209  // Allocate and build uniform rectangular grid
210  getRegularGrid(gridLimitsLow, gridLimitsUp);
211

277 size_t
278 RegularGrid::getBoxMembership(const gsl_vector *state) const
279 {
280  const size_t dim = state->size;
281  size_t nFoundInt, nBoxDir, foundBox, idInt;
282  gsl_vector *boundsDir;
283  gsl_vector_uint *multiIdx = gsl_vector_uint_alloc(dim);
284 
285  // Get box
286  foundBox = N;
287  nFoundInt = 0;
288  // Check for each dimension
289  for (size_t d = 0; d < dim; d++)
290  {
291  // Get bounds for each interval of direction d
292  boundsDir = bounds->at(d);
293  // Get the number of intervals
294  nBoxDir = gsl_vector_uint_get(nx, d);
295 
296  // If state outisde domain, no need to continue to test,
297  // the box is not found.
298  // The intervals must not overlap and be ordered increasingly.
299  idInt = 0;
300  if (gsl_vector_get(state, d) < gsl_vector_get(boundsDir, idInt))
301  break;
302 
303  // Get interval for direction d
304  while ((idInt < nBoxDir) && (gsl_vector_get(state, d)
305  >= gsl_vector_get(boundsDir, idInt + 1)))
306  // Not in interval idInt, iterate to next interval
307  idInt++;
308 
309  // Check if interval was found.
310  // If yes, save the index for direction d and go to next direction.
311  // Otherwise the box is outside the domain, no need to continue,
312  // the box is not found.
313  if (idInt < nBoxDir)
314  {
315  nFoundInt++;
316  gsl_vector_uint_set(multiIdx, d, idInt);
317  continue;
318  }
319  else
320  break;
321  }
322 
323  // Is the state in this box? If yes, flatten index
324  if (nFoundInt == dim)
325  foundBox = ravel_multi_index(multiIdx, nx);
326 
327  // Free
328  gsl_vector_uint_free(multiIdx);
329 
330  return foundBox;
331 }
