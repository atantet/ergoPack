131 void
132 RegularGrid::getRegularGrid(const gsl_vector *xmin, const gsl_vector *xmax)
133 {
134  double delta;
135 
136  // Build uniform grid bounds
137  for (size_t d = 0; d < dim; d++)
138  {
139  // Get spatial step
140  delta = (gsl_vector_get(xmax, d) - gsl_vector_get(xmin, d))
141  / gsl_vector_uint_get(nx, d);
142 
143  // Set grid bounds
144  gsl_vector_set((*bounds)[d], 0, gsl_vector_get(xmin, d));
145  for (size_t i = 1; i < gsl_vector_uint_get(nx, d) + 1; i++)
146  {
147  gsl_vector_set((*bounds)[d], i,
148  gsl_vector_get((*bounds)[d], i-1) + delta);
149  }
150  }

164 RegularGrid::getAdaptedRegularGrid(const gsl_vector *nSTDLow, const gsl_vector *nSTDHigh,
165  const gsl_matrix *states)
166 {
167  gsl_vector *xmin, *xmax, *statesMean, *statesSTD;
168 
169  xmin = gsl_vector_alloc(dim);
170  xmax = gsl_vector_alloc(dim);
171 
172  // If then number of STD is given, get mean and std of time series
173  // Otherwise, get minimum and maximum
174  if (nSTDLow && nSTDHigh)
175  {
176  // Find mean and std
177  statesMean = gsl_vector_alloc(states->size2);
178  statesSTD = gsl_vector_alloc(states->size2);
179  gsl_matrix_get_mean(statesMean, states, 0);
180  gsl_matrix_get_std(statesSTD, states, 0);
181 
182  // Define limits as multiples of std around the mean
183  for (size_t d = 0; d < dim; d++)
184  {
185  gsl_vector_set(xmin, d, gsl_vector_get(statesMean, 0)
186  - gsl_vector_get(nSTDLow, d)
187  * gsl_vector_get(statesSTD, d));
188  gsl_vector_set(xmax, d, gsl_vector_get(statesMean, d)
189  + gsl_vector_get(nSTDHigh, d)
190  * gsl_vector_get(statesSTD, d));
191  }
192 
193  // Free
194  gsl_vector_free(statesMean);
195  gsl_vector_free(statesSTD);
196  }
197  else
198  {
199  // Get min and max of time series
200  gsl_matrix_get_min(xmin, states, 0);
201  gsl_matrix_get_max(xmax, states, 0);
202  // The upper bound uses a strict inequality,
203  // make sure the maximum is included
204  gsl_vector_add_constant(xmax, 1.e-12);
205  }
206 
207  // Allocate and build uniform rectangular grid
208  getRegularGrid(xmin, xmax);
209 
210  // Free
211  gsl_vector_free(xmin);
212  gsl_vector_free(xmax);
