649 gsl_spmatrix2AR::MultMv(double *v, double *w)
650 {
651  size_t j, outerIdx, p, n;
652 
653  /* form y := 0 */
654  for (j = 0; j < M->size1; ++j)
655  {
656  w[j] = 0.0;
657  }
658 
659  /* form w := M * v */
660  if (GSL_SPMATRIX_ISCRS(M))
661  {
662  /* (row, column) = (outerIdx, M->i) */
663  for (outerIdx = 0; outerIdx < M->outerSize; ++outerIdx)
664  {
665  for (p = M->p[outerIdx]; p < M->p[outerIdx + 1]; ++p)
666  {
667  w[outerIdx] += M->data[p] * v[M->i[p]];
668  }
669  }
670  }
671  else if (GSL_SPMATRIX_ISCCS(M))
672  {
673  /* (row, column) = (M->i, outerIdx) */
674  for (outerIdx = 0; outerIdx < M->outerSize; ++outerIdx)
675  {
676  for (p = M->p[outerIdx]; p < M->p[outerIdx + 1]; ++p)
677  {
678  w[M->i[p]] += M->data[p] * v[outerIdx];
679  }
680  }
681  }
682  else if (GSL_SPMATRIX_ISTRIPLET(M))
683  {
684  /* (row, column) = (M->i, M->p) */
685  for (n = 0; n < M->nz; ++n)
686  {
687  w[M->i[n]] += M->data[n] * v[M->p[n]];
688  }
689  }

709 void
710 getSpectrumAR(const int nev, ARNonSymStdEig<double, gsl_spmatrix2AR > *EigProb,
711  gsl_spmatrix2AR *gsl2AR, configAR cfgAR,
712  gsl_vector_complex *EigVal, gsl_matrix_complex *EigVec)
713 {
714  size_t N = EigVec->size1;
715  gsl_complex element;
716 
717  // Allocate vectors given to ARPACK++
718  double *EigValReal = new double [nev+1];
719  double *EigValImag = new double [nev+1];
720  double *EigVecRealImag = new double [(nev+2) * N];
721 
723  EigProb->DefineParameters(gsl2AR->M->size1, nev, gsl2AR,
724  &gsl_spmatrix2AR::MultMv,
725  cfgAR.which, cfgAR.ncv, cfgAR.tol,
726  cfgAR.maxit, cfgAR.resid, cfgAR.AutoShift);
727 
729  EigProb->EigenValVectors(EigVecRealImag, EigValReal, EigValImag);
730 
732  for (size_t ev = 0; ev < (size_t) nev; ev++)
733  {
734  element = gsl_complex_rect(EigValReal[ev], EigValImag[ev]);
735  gsl_vector_complex_set(EigVal, ev, element);
736 
737  // Add real part of eigenvector
738  for (size_t i = 0; i < N; i++)
739  {
740  element = gsl_complex_rect(EigVecRealImag[ev*N + i], 0.);
741  gsl_matrix_complex_set(EigVec, i, ev, element);
742  }
743 
744  // If complex pair
745  if ((gsl_pow_2(EigValImag[ev]) > 1.e-12) && (ev + 1 < nev))
746  {
747  // Add complex conjugate eigenvalue
748  element = gsl_complex_conjugate(gsl_vector_complex_get(EigVal, ev));
749  gsl_vector_complex_set(EigVal, ev + 1, element);
750 
751  // Add imaginary part to eigenvector
752  for (size_t i = 0; i < N; i++)
753  {
754  element = gsl_complex_rect(GSL_REAL(gsl_matrix_complex_get(EigVec, i, ev)),
755  EigVecRealImag[(ev + 1)*N + i]);
756  gsl_matrix_complex_set(EigVec, i, ev, element);
757  }
758 
759  // Add complex conjugate eigenvector
760  for (size_t i = 0; i < N; i++)
761  {
762  element = gsl_complex_conjugate(gsl_matrix_complex_get(EigVec, i, ev));
763  gsl_matrix_complex_set(EigVec, i, ev + 1, element);
764  }
765 
766  // Increment eigenvalue one more time
767  ev++;
768  }
769  }
770 
771  //Free
772  delete[] EigValReal;
773  delete[] EigValImag;
774  delete[] EigVecRealImag;
775 
776  return;
777 }
