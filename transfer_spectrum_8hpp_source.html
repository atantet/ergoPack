342 gsl_spmatrix2AR::MultMv(double *v, double *w)
343 {
344  size_t j, outerIdx, p, n;
345 
346  /* form y := 0 */
347  for (j = 0; j < M->size1; ++j)
348  {
349  w[j] = 0.0;
350  }
351 
352  /* form w := M * v */
353  if (GSL_SPMATRIX_ISCRS(M))
354  {
355  /* (row, column) = (outerIdx, M->i) */
356  for (outerIdx = 0; outerIdx < M->outerSize; ++outerIdx)
357  {
358  for (p = M->p[outerIdx]; p < M->p[outerIdx + 1]; ++p)
359  {
360  w[outerIdx] += M->data[p] * v[M->i[p]];
361  }
362  }
363  }
364  else if (GSL_SPMATRIX_ISCCS(M))
365  {
366  /* (row, column) = (M->i, outerIdx) */
367  for (outerIdx = 0; outerIdx < M->outerSize; ++outerIdx)
368  {
369  for (p = M->p[outerIdx]; p < M->p[outerIdx + 1]; ++p)
370  {
371  w[M->i[p]] += M->data[p] * v[outerIdx];
372  }
373  }
374  }
375  else if (GSL_SPMATRIX_ISTRIPLET(M))
376  {
377  /* (row, column) = (M->i, M->p) */
378  for (n = 0; n < M->nz; ++n)
379  {
380  w[M->i[n]] += M->data[n] * v[M->p[n]];
381  }
382  }
383 
384  return;

483 void
484 writeSpectrumAR(FILE *fEigVal, FILE *fEigVec,
485  const double *EigValReal, const double *EigValImag,
486  const double *EigVec, const int nev, const size_t N)
487 {
488  size_t vecCount = 0;
489  int ev =0;
490  // Write real and imaginary parts of each eigenvalue on each line
491  // Write on each pair of line the real part of an eigenvector then its imaginary part
492  while (ev < nev) {
493  // Always write the eigenvalue
494  fprintf(fEigVal, "%lf %lf\n", EigValReal[ev], EigValImag[ev]);
495  // Always write the real part of the eigenvector ev
496  for (size_t i = 0; i < N; i++){
497  fprintf(fEigVec, "%lf ", EigVec[vecCount*N+i]);
498  }
499  fprintf(fEigVec, "\n");
500  vecCount++;
501 
502  // Write its imaginary part or the zero vector
503  if (EigValImag[ev] != 0.){
504  for (size_t i = 0; i < N; i++)
505  fprintf(fEigVec, "%lf ", EigVec[vecCount*N+i]);
506  vecCount++;
507  // Skip the conjugate
508  ev += 2;
509  }
510  else{
511  for (size_t i = 0; i < N; i++)
512  fprintf(fEigVec, "%lf ", 0.);
513  ev += 1;
514  }
515  fprintf(fEigVec, "\n");
516  }
517 
519  if (ferror(fEigVal))
520  {
521  throw std::ios::failure("writeSpectrumAR, printing eigenvalues");
522  }
523  if (ferror(fEigVec))
524  {
525  throw std::ios::failure("writeSpectrumAR, printing eigenvectors");
526  }
527 
528  return;
