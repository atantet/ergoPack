542 gsl_spmatrix2AR::MultMv(double *v, double *w)
543 {
544  size_t j, outerIdx, p, n;
545 
546  /* form y := 0 */
547  for (j = 0; j < M->size1; ++j)
548  {
549  w[j] = 0.0;
550  }
551 
552  /* form w := M * v */
553  if (GSL_SPMATRIX_ISCRS(M))
554  {
555  /* (row, column) = (outerIdx, M->i) */
556  for (outerIdx = 0; outerIdx < M->size1; ++outerIdx)
557  {
558  for (p = M->p[outerIdx]; p < M->p[outerIdx + 1]; ++p)
559  {
560  w[outerIdx] += M->data[p] * v[M->i[p]];
561  }
562  }
563  }
564  else if (GSL_SPMATRIX_ISCCS(M))
565  {
566  /* (row, column) = (M->i, outerIdx) */
567  for (outerIdx = 0; outerIdx < M->size2; ++outerIdx)
568  {
569  for (p = M->p[outerIdx]; p < M->p[outerIdx + 1]; ++p)
570  {
571  w[M->i[p]] += M->data[p] * v[outerIdx];
572  }
573  }
574  }
575  else if (GSL_SPMATRIX_ISTRIPLET(M))
576  {
577  /* (row, column) = (M->i, M->p) */
578  for (n = 0; n < M->nz; ++n)
579  {
580  w[M->i[n]] += M->data[n] * v[M->p[n]];
581  }
582  }
583 
584  return;
585 }

604 void
605 getSpectrumAR(int *nev, const size_t N,
606  ARNonSymStdEig<double, gsl_spmatrix2AR > *EigProb,
607  gsl_spmatrix2AR *gsl2AR, configAR cfgAR,
608  gsl_vector_complex **EigVal, gsl_matrix_complex **EigVec)
609 {
610  gsl_complex element;
611 
612  // Allocate vectors given to ARPACK++
613  double *EigValReal = new double [*nev+1];
614  double *EigValImag = new double [*nev+1];
615  double *EigVecRealImag = new double [(*nev+2) * N];
616 
618  EigProb->DefineParameters(gsl2AR->M->size1, *nev, gsl2AR,
619  &gsl_spmatrix2AR::MultMv,
620  cfgAR.which, cfgAR.ncv, cfgAR.tol,
621  cfgAR.maxit, cfgAR.resid, cfgAR.AutoShift);
622 
624  EigProb->EigenValVectors(EigVecRealImag, EigValReal, EigValImag);
625 
627  *nev = EigProb->ConvergedEigenvalues();
628 
630  *EigVal = gsl_vector_complex_alloc(*nev);
632  *EigVec = gsl_matrix_complex_alloc(*nev, N);
633 
635  for (size_t ev = 0; ev < (size_t) *nev; ev++)
636  {
637  element = gsl_complex_rect(EigValReal[ev], EigValImag[ev]);
638  gsl_vector_complex_set(*EigVal, ev, element);
639 
640  // Add real part of eigenvector
641  for (size_t i = 0; i < N; i++)
642  {
643  element = gsl_complex_rect(EigVecRealImag[ev*N + i], 0.);
644  gsl_matrix_complex_set(*EigVec, ev, i, element);
645  }
646 
647  // If complex pair
648  if ((gsl_pow_2(EigValImag[ev]) > 1.e-12) && (ev + 1 < *nev))
649  {
650  // Add complex conjugate eigenvalue
651  element = gsl_complex_conjugate(gsl_vector_complex_get(*EigVal, ev));
652  gsl_vector_complex_set(*EigVal, ev + 1, element);
653 
654  // Add imaginary part to eigenvector
655  for (size_t i = 0; i < N; i++)
656  {
657  element = gsl_complex_rect(GSL_REAL(gsl_matrix_complex_get(*EigVec,
658  ev, i)),
659  EigVecRealImag[(ev + 1)*N + i]);
660  gsl_matrix_complex_set(*EigVec, ev, i, element);
661  }
662 
663  // Add complex conjugate eigenvector
664  for (size_t i = 0; i < N; i++)
665  {
666  element = gsl_complex_conjugate(gsl_matrix_complex_get(*EigVec, ev, i));
667  gsl_matrix_complex_set(*EigVec, ev + 1, i, element);
668  }
669 
670  // Increment eigenvalue one more time
671  ev++;
672  }
673  }
674 
675  //Free
676  delete[] EigValReal;
677  delete[] EigValImag;
678  delete[] EigVecRealImag;
679 
680  return;
