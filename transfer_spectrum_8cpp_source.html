480 void
481 gsl_spmatrix2AR::MultMv(double *v, double *w)
482 {
483  size_t j, outerIdx, p, n;
484 
485  /* form y := 0 */
486  for (j = 0; j < M->size1; ++j)
487  {
488  w[j] = 0.0;
489  }
490 
491  /* form w := M * v */
492  if (GSL_SPMATRIX_ISCRS(M))
493  {
494  /* (row, column) = (outerIdx, M->i) */
495  for (outerIdx = 0; outerIdx < M->size1; ++outerIdx)
496  {
497  for (p = M->p[outerIdx]; p < M->p[outerIdx + 1]; ++p)
498  {
499  w[outerIdx] += M->data[p] * v[M->i[p]];
500  }
501  }
502  }
503  else if (GSL_SPMATRIX_ISCCS(M))
504  {
505  /* (row, column) = (M->i, outerIdx) */
506  for (outerIdx = 0; outerIdx < M->size2; ++outerIdx)
507  {
508  for (p = M->p[outerIdx]; p < M->p[outerIdx + 1]; ++p)
509  {
510  w[M->i[p]] += M->data[p] * v[outerIdx];
511  }
512  }
513  }
514  else if (GSL_SPMATRIX_ISTRIPLET(M))
515  {
516  /* (row, column) = (M->i, M->p) */
517  for (n = 0; n < M->nz; ++n)
518  {
519  w[M->i[n]] += M->data[n] * v[M->p[n]];
520  }
521  }
522 
523  return;
524 }

541 void
542 getSpectrumAR(const int nev, ARNonSymStdEig<double, gsl_spmatrix2AR > *EigProb,
543  gsl_spmatrix2AR *gsl2AR, configAR cfgAR,
544  gsl_vector_complex *EigVal, gsl_matrix_complex *EigVec)
545 {
546  size_t N = EigVec->size1;
547  gsl_complex element;
548 
549  // Allocate vectors given to ARPACK++
550  double *EigValReal = new double [nev+1];
551  double *EigValImag = new double [nev+1];
552  double *EigVecRealImag = new double [(nev+2) * N];
553 
555  EigProb->DefineParameters(gsl2AR->M->size1, nev, gsl2AR,
556  &gsl_spmatrix2AR::MultMv,
557  cfgAR.which, cfgAR.ncv, cfgAR.tol,
558  cfgAR.maxit, cfgAR.resid, cfgAR.AutoShift);
559 
561  EigProb->EigenValVectors(EigVecRealImag, EigValReal, EigValImag);
562 
564  for (size_t ev = 0; ev < (size_t) nev; ev++)
565  {
566  element = gsl_complex_rect(EigValReal[ev], EigValImag[ev]);
567  gsl_vector_complex_set(EigVal, ev, element);
568 
569  // Add real part of eigenvector
570  for (size_t i = 0; i < N; i++)
571  {
572  element = gsl_complex_rect(EigVecRealImag[ev*N + i], 0.);
573  gsl_matrix_complex_set(EigVec, i, ev, element);
574  }
575 
576  // If complex pair
577  if ((gsl_pow_2(EigValImag[ev]) > 1.e-12) && (ev + 1 < nev))
578  {
579  // Add complex conjugate eigenvalue
580  element = gsl_complex_conjugate(gsl_vector_complex_get(EigVal, ev));
581  gsl_vector_complex_set(EigVal, ev + 1, element);
582 
583  // Add imaginary part to eigenvector
584  for (size_t i = 0; i < N; i++)
585  {
586  element = gsl_complex_rect(GSL_REAL(gsl_matrix_complex_get(EigVec, i, ev)),
587  EigVecRealImag[(ev + 1)*N + i]);
588  gsl_matrix_complex_set(EigVec, i, ev, element);
589  }
590 
591  // Add complex conjugate eigenvector
592  for (size_t i = 0; i < N; i++)
593  {
594  element = gsl_complex_conjugate(gsl_matrix_complex_get(EigVec, i, ev));
595  gsl_matrix_complex_set(EigVec, i, ev + 1, element);
596  }
597 
598  // Increment eigenvalue one more time
599  ev++;
600  }
601  }
602 
603  //Free
604  delete[] EigValReal;
605  delete[] EigValImag;
606  delete[] EigVecRealImag;
607 
608  return;
609 }
