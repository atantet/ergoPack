321 {
322  // Assign pointers to workspace vectors
323  gsl_vector_view tmp = gsl_matrix_row(work, 0);
324  gsl_vector_view tmp1 = gsl_matrix_row(work, 1);
325  gsl_vector_view presentState;

328  delayedField->evalField(currentState, &tmp.vector);
329  // Scale by time step

331

335  // Assign pointer to present state

337

339  stocField->evalField(&presentState.vector, &tmp1.vector);
340  // Scale by time step

342 
343  // Add drift to present state
