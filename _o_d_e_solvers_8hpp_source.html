686

689  gsl_vector_scale(&k2.vector, dt);

692  gsl_vector_scale(&tmp.vector, 0.5);
693  gsl_vector_add(&tmp.vector, currentState);
694

697  gsl_vector_scale(&k3.vector, dt);
698

700  gsl_vector_add(&tmp.vector, currentState);
701 
702  // Fourth increment
703  field->evalField(&tmp.vector, &k4.vector);
704  gsl_vector_scale(&k4.vector, dt);
