683 {
684  const size_t nTraj = gridMem->size1;
685  size_t box0, boxf;
686  size_t nOut = 0;
687  gsl_spmatrix *T;
688 
690  if (!(T = gsl_spmatrix_alloc_nzmax(N, N, nTraj, GSL_SPMATRIX_TRIPLET)))
691  {
692  std::bad_alloc();
693  }
694 
696  for (size_t traj = 0; traj < nTraj; traj++) {
697  box0 = gsl_matrix_uint_get(gridMem, traj, 0);
698  boxf = gsl_matrix_uint_get(gridMem, traj, 1);
699 
701  if ((box0 < N) && (boxf < N))
702  gsl_spmatrix_set(T, box0, boxf, 1., 1);
703  else
704  nOut++;
705  }
706  std::cout << nOut * 100. / nTraj
707  << "% of the trajectories ended up out of the domain." << std::endl;
708 
709  return T;

728 filterStochasticMatrix(gsl_spmatrix *M,
729  gsl_vector *rowCut, gsl_vector *colCut,
730  double tol, int norm)
731 {
732  size_t outerIdx, p, n;
733  size_t isRowOut, isColOut;
734  double totalSum;
735  gsl_vector *sum;
736  gsl_vector_uint *rowOut, *colOut;
737 
740  rowOut = gsl_vector_uint_calloc(M->size1);
741  for (n = 0; n < M->size1; n++)
742  {
743  if (gsl_vector_get(rowCut, n) < tol)
744  {
745  gsl_vector_uint_set(rowOut, n, 1);
746  gsl_vector_set(rowCut, n, 0.);
747  }
748  }
749 
750  colOut = gsl_vector_uint_calloc(M->size2);
751  for (n = 0; n < M->size2; n++)
752  {
753  if (gsl_vector_get(colCut, n) < tol)
754  {
755  gsl_vector_uint_set(colOut, n, 1);
756  gsl_vector_set(colCut, n, 0.);
757  }
758  }
759 
760  if (GSL_SPMATRIX_ISTRIPLET(M))
761  {
762  GSL_ERROR("first index out of range", GSL_EINVAL);
763  }
764  else if (GSL_SPMATRIX_ISCCS(M))
765  {
766  for (outerIdx = 0; outerIdx < M->outerSize; outerIdx++)
767  {
768  isColOut = gsl_vector_uint_get(colOut, outerIdx);
769  for (p = M->p[outerIdx]; p < M->p[outerIdx + 1]; p++)
770  {
771  isRowOut = gsl_vector_uint_get(rowOut, M->i[p]);
772  // Remove elements of states to be removed
773  if (isRowOut || isColOut)
774  M->data[p] = 0.;
775  }
776  }
777  }
778  else if (GSL_SPMATRIX_ISCRS(M))
779  {
780  for (outerIdx = 0; outerIdx < M->outerSize; outerIdx++)
781  {
782  isRowOut = gsl_vector_uint_get(rowOut, outerIdx);
783  for (p = M->p[outerIdx]; p < M->p[outerIdx + 1]; p++)
784  {
785  isColOut = gsl_vector_uint_get(colOut, M->i[p]);
786  // Remove elements of states to be removed
787  if (isRowOut || isColOut)
788  M->data[p] = 0.;
789  }
790  }
791  }
792 
794  switch (norm)
795  {
796  case 0:
797  totalSum = gsl_spmatrix_get_sum(M);
798  gsl_spmatrix_scale(M, 1. / totalSum);
799  break;
800  case 1:
801  sum = gsl_vector_alloc(M->size1);
802  gsl_spmatrix_get_rowsum(sum, M);
803  gsl_spmatrix_div_rows(M, sum);
804  gsl_vector_free(sum);
805  break;
806  case 2:
807  sum = gsl_vector_alloc(M->size1);
808  gsl_spmatrix_get_colsum(sum, M);
809  gsl_spmatrix_div_cols(M, sum);
810  gsl_vector_free(sum);
811  break;
812  default:
813  break;
814  }
815  gsl_vector_normalize(rowCut);
816  gsl_vector_normalize(colCut);
817 
819  gsl_vector_uint_free(rowOut);
820  gsl_vector_uint_free(colOut);
821 
822  return 0;
823 }
