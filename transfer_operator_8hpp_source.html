666 int
667 filterStochasticMatrix(gsl_spmatrix *M,
668  gsl_vector *rowCut, gsl_vector *colCut,
669  double tol, int norm)
670 {
671  size_t outerIdx, p, n;
672  size_t isRowOut, isColOut;
673  double totalSum;
674  gsl_vector *sum;
675  gsl_vector_uint *rowOut, *colOut;
676 
679  rowOut = gsl_vector_uint_calloc(M->size1);
680  for (n = 0; n < M->size1; n++)
681  {
682  if (gsl_vector_get(rowCut, n) < tol)
683  {
684  gsl_vector_uint_set(rowOut, n, 1);
685  gsl_vector_set(rowCut, n, 0.);
686  }
687  }
688 
689  colOut = gsl_vector_uint_calloc(M->size2);
690  for (n = 0; n < M->size2; n++)
691  {
692  if (gsl_vector_get(colCut, n) < tol)
693  {
694  gsl_vector_uint_set(colOut, n, 1);
695  gsl_vector_set(colCut, n, 0.);
696  }
697  }
698 
699  if (GSL_SPMATRIX_ISTRIPLET(M))
700  {
701  GSL_ERROR("first index out of range", GSL_EINVAL);
702  }
703  else if (GSL_SPMATRIX_ISCCS(M))
704  {
705  for (outerIdx = 0; outerIdx < M->outerSize; outerIdx++)
706  {
707  isColOut = gsl_vector_uint_get(colOut, outerIdx);
708  for (p = M->p[outerIdx]; p < M->p[outerIdx + 1]; p++)
709  {
710  isRowOut = gsl_vector_uint_get(rowOut, M->i[p]);
711  // Remove elements of states to be removed
712  if (isRowOut || isColOut)
713  M->data[p] = 0.;
714  }
715  }
716  }
717  else if (GSL_SPMATRIX_ISCRS(M))
718  {
719  for (outerIdx = 0; outerIdx < M->outerSize; outerIdx++)
720  {
721  isRowOut = gsl_vector_uint_get(rowOut, outerIdx);
722  for (p = M->p[outerIdx]; p < M->p[outerIdx + 1]; p++)
723  {
724  isColOut = gsl_vector_uint_get(colOut, M->i[p]);
725  // Remove elements of states to be removed
726  if (isRowOut || isColOut)
727  M->data[p] = 0.;
728  }
729  }
730  }
731 
733  switch (norm)
734  {
735  case 0:
736  totalSum = gsl_spmatrix_get_sum(M);
737  gsl_spmatrix_scale(M, 1. / totalSum);
738  break;
739  case 1:
740  sum = gsl_spmatrix_get_rowsum(M);
741  gsl_spmatrix_div_rows(M, sum);
742  gsl_vector_free(sum);
743  break;
744  case 2:
745  sum = gsl_spmatrix_get_colsum(M);
746  gsl_spmatrix_div_cols(M, sum);
747  gsl_vector_free(sum);
748  break;
749  default:
750  break;
751  }
752  gsl_vector_normalize(rowCut);
753  gsl_vector_normalize(colCut);
754 
756  gsl_vector_uint_free(rowOut);
757  gsl_vector_uint_free(colOut);
758 
759  return 0;
760 }
