428 RegularGrid::getAdaptedRegularGrid(const gsl_vector *nSTDLow, const gsl_vector *nSTDHigh,
429  const gsl_matrix *states)
430 {
431  gsl_vector *xmin, *xmax, *statesMean, *statesSTD;
432 
433  xmin = gsl_vector_alloc(dim);
434  xmax = gsl_vector_alloc(dim);
435 
436  // If then number of STD is given, get mean and std of time series
437  // Otherwise, get minimum and maximum
438  if (nSTDLow && nSTDHigh)
439  {
440  // Find mean and std
441  statesMean = gsl_vector_alloc(states->size2);
442  statesSTD = gsl_vector_alloc(states->size2);
443  gsl_matrix_get_mean(statesMean, states, 0);
444  gsl_matrix_get_std(statesSTD, states, 0);
445 
446  // Define limits as multiples of std around the mean
447  for (size_t d = 0; d < dim; d++)
448  {
449  gsl_vector_set(xmin, d, gsl_vector_get(statesMean, 0)
450  - gsl_vector_get(nSTDLow, d)
451  * gsl_vector_get(statesSTD, d));
452  gsl_vector_set(xmax, d, gsl_vector_get(statesMean, d)
453  + gsl_vector_get(nSTDHigh, d)
454  * gsl_vector_get(statesSTD, d));
455  }
456 
457  // Free
458  gsl_vector_free(statesMean);
459  gsl_vector_free(statesSTD);
460  }
461  else
462  {
463  // Get min and max of time series
464  gsl_matrix_get_min(xmin, states, 0);
465  gsl_matrix_get_max(xmax, states, 0);
466  // The upper bound uses a strict inequality,
467  // make sure the maximum is included
468  gsl_vector_add_constant(xmax, 1.e-12);
469  }
470 
471  // Allocate and build uniform rectangular grid
472  getRegularGrid(xmin, xmax);
473 
474  // Free
475  gsl_vector_free(xmin);
476  gsl_vector_free(xmax);

541 {
542  const size_t dim = state->size;
543  size_t inBox, nBoxDir;
544  size_t foundBox;
545  size_t subbp, subbn, ids;
546  gsl_vector *dimBounds;
547 
548  // Get box
549  foundBox = N;
550  for (size_t box = 0; box < N; box++)
551  {
552  inBox = 0;
553  subbp = box;
554  for (size_t d = 0; d < dim; d++)
555  {
556  dimBounds = bounds->at(d);
557  nBoxDir = dimBounds->size - 1;
558  subbn = (size_t) (subbp / nBoxDir);
559  ids = subbp - subbn * nBoxDir;
560  inBox += (size_t) ((gsl_vector_get(state, d)
561  >= gsl_vector_get(dimBounds, ids))
562  & (gsl_vector_get(state, d)
563  < gsl_vector_get(dimBounds, ids+1)));
564  subbp = subbn;
565  }
566  if (inBox == dim)
567  {
568  foundBox = box;
569  break;
570  }
571  }
572 
573  return foundBox;

660 PolarGrid::getLevelsMembership()
661 {
662  // Allocate vectors
663  const size_t NSupport = supportGrid->getN();
664  gsl_vector *levelsDensity = gsl_vector_calloc(nLevels);
665  gsl_vector *targetDensity = gsl_vector_alloc(nLevels);
666  gsl_vector_uint *isNotUsed = gsl_vector_uint_alloc(NSupport);
667  size_t idxMaxDenAmongNotUsed;
668  size_t nUsed = 0;
669  gsl_vector_uint *supportLevelsMem = gsl_vector_uint_calloc(NSupport);
670 
671  // Set vectors
672  gsl_vector_set_all(targetDensity, 1. / nLevels);
673  gsl_vector_uint_set_all(isNotUsed, 1);
674 
675  // Get levels membership
676  for (size_t lev = 0; lev < nLevels; lev++)
677  {
678  while ((gsl_vector_get(levelsDensity, lev)
679  < gsl_vector_get(targetDensity, lev))
680  && (nUsed < NSupport))
681  {
682  // Get nonused box with maximum density
683  idxMaxDenAmongNotUsed \
684  = gsl_vector_max_index_among(supportDensity, isNotUsed);
685 
686  // Add this box to level
687  gsl_vector_uint_set(supportLevelsMem, idxMaxDenAmongNotUsed, lev);
688 
689  // Add the density of this box to the level
690  gsl_vector_set(levelsDensity, lev, gsl_vector_get(levelsDensity, lev)
691  + gsl_vector_get(supportDensity, idxMaxDenAmongNotUsed));
692 
693  // Mark the box as used
694  gsl_vector_uint_set(isNotUsed, idxMaxDenAmongNotUsed, 0);
695 
696  // Increment the number of used boxes
697  nUsed++;
698  }
699  }
700

702  gsl_vector_free(targetDensity);
703  gsl_vector_free(levelsDensity);
704  gsl_vector_uint_free(isNotUsed);

733  E = gsl_matrix_alloc(dim, dim);
