631  size_t idxMaxDenAmongNotUsed;

633  gsl_vector_uint *supportLevelsMem = gsl_vector_uint_calloc(NSupport);
634

637  gsl_vector_uint_set_all(isNotUsed, 1);
638 
639  // Get levels membership
640  for (size_t lev = 0; lev < nLevels; lev++)
641  {
642  while ((gsl_vector_get(levelsDensity, lev)
643  < gsl_vector_get(targetDensity, lev))

645  {

648  = gsl_vector_max_index_among(supportDensity, isNotUsed);
